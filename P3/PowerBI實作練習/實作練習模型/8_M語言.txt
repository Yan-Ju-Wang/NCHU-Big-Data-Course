------------------------------------------------------------------------------
獎金
=Table.Last(
    Table.SelectRows(獎金比例對照表,(x)=>[交易金額]>=x[等級]))[獎金比例]*[交易金額]

愛心捐款
=Table.Last(
    Table.SelectRows(獎金比例對照表,(x)=>[交易金額]>=x[等級]))[愛心捐款]

福利金
=Table.Last(
    Table.SelectRows(獎金比例對照表,(x)=>[交易金額]>=x[等級]))[福利金]

淨得
=[交易金額]-List.Sum({[獎金],[福利金],[愛心捐款]})
